        // Eksempeldata for radiostasjoner
        const stationsData = {
            NRK: [
                { name: "NRK P1 Sørlandet", url: "https://lyd.nrk.no/nrk_radio_p1_sorlandet_mp3_m.xspf", logo: "nrklogo/sorlandet.jpeg" },
                { name: "NRK P1 Rogaland", url: "https://lyd.nrk.no/nrk_radio_p1_rogaland_mp3_h", logo: "nrklogo/rogaland.jpeg" },
                { name: "NRK P1 Østfold", url: "https://lyd.nrk.no/nrk_radio_p1_ostfold_mp3_m.xspf", logo: "nrklogo/ostfold.jpeg" },                
                { name: "NRK P2", url: "https://lyd.nrk.no/nrk_radio_p2_mp3_h", logo: "nrklogo/logo2.png" },
                { name: "NRK P3", url: "https://lyd.nrk.no/nrk_radio_p3_mp3_h", logo: "nrklogo/logo3.png" },
                { name: "NRK Alltid Nyheter", url: "https://lyd.nrk.no/nrk_radio_alltidnyheter_mp3_h", logo: "nrklogo/logo4.png" }
            ],
            P4: [
                { name: "P4 Direkte", url: "https://p4-direkte-url", logo: "path/to/logo5.png" },
                { name: "P4 Hits", url: "https://p4-hits-url", logo: "path/to/logo6.png" }
            ]
            // Legg til flere kategorier og stasjoner som ønsket
        };

        let currentAudio = null;
        let currentlyPlayingUrl = null;




function showStations(category) {
    const stationsGrid = document.getElementById('stations-grid');
    stationsGrid.innerHTML = ''; // Tømmer nåværende innhold
    stationsData[category].forEach(station => {
        const stationDiv = document.createElement('div');
        stationDiv.classList.add('station');
        stationDiv.onclick = () => togglePlayStation(station.url, stationDiv);

        // Legg til logo
        const logo = document.createElement('img');
        logo.src = station.logo;
        logo.alt = `${station.name} logo`;
        stationDiv.appendChild(logo);

        // Stasjonsnavn
        const name = document.createElement('h3');
        name.textContent = station.name;
        stationDiv.appendChild(name);

        // Sang- og stasjonsinformasjon
        const songInfo = document.createElement('p');
        songInfo.id = `${station.name}-song-info`; // Unik ID for hver stasjon
        songInfo.textContent = "Laster...";
        stationDiv.appendChild(songInfo);

        stationsGrid.appendChild(stationDiv);
    });
}

        // Funksjon for å vise stasjoner basert på valgt kategori


        // Funksjon for å spille av eller stoppe en stasjon
        function togglePlayStation(url, stationDiv) {
            // Hvis vi trykker på samme stasjon som spiller, stopp avspillingen
            if (currentAudio && url === currentlyPlayingUrl) {
                currentAudio.pause();
                currentlyPlayingUrl = null; // Nullstill for å indikere at ingenting spiller
                stationDiv.classList.remove('playing'); // Fjern visuell indikasjon for aktiv stasjon
            } else {
                // Hvis en annen stasjon spiller, stopp den
                if (currentAudio) {
                    currentAudio.pause();
                }

                // Start den nye stasjonen
                currentAudio = new Audio(url);
                currentAudio.play();
                currentlyPlayingUrl = url; // Sett den nye URL-en som aktiv

                // Fjern 'playing'-klassen fra alle stasjonsbokser
                document.querySelectorAll('.station').forEach(st => st.classList.remove('playing'));
                stationDiv.classList.add('playing'); // Legg til visuell indikasjon for aktiv stasjon
            }
        }

        // Last inn første kategori automatisk
        showStations('NRK');
    </script>
<script>
        const xspfUrl = 'https://lyd.nrk.no/nrk_radio_p1_sorlandet_mp3_l.xspf';
        let currentStreamUrl = ''; // Holder på gjeldende strøm-URL for sammenligning

        async function loadRadioStream() {
            console.log('Starter innlasting av XSPF-fil for metadata...');
            try {
                // Henter XSPF-filen
                const response = await fetch(xspfUrl);
                console.log('Forespørsel sendt til:', xspfUrl);

                if (!response.ok) {
                    console.error('Kunne ikke laste XSPF-filen:', response.status);
                    throw new Error('Kunne ikke laste XSPF-filen.');
                }

                // Leser innholdet som tekst
                const xspfContent = await response.text();
                console.log('XSPF-innhold mottatt:', xspfContent);

                // Parser XML-innholdet med namespace
                const parser = new DOMParser();
                const xmlDoc = parser.parseFromString(xspfContent, 'application/xml');
                const namespace = 'http://xspf.org/ns/0/';

                // Finner <track> elementene i XML
                const trackElement = xmlDoc.getElementsByTagNameNS(namespace, 'track')[0];
                if (trackElement) {
                    // Henter strøm-URL fra <location>
                    const streamUrl = trackElement.getElementsByTagNameNS(namespace, 'location')[0]?.textContent.trim();
                    console.log('Stream URL funnet:', streamUrl);

                    // Henter program/sangtittel fra <title> i track-elementet
                    const programTitleElement = trackElement.getElementsByTagNameNS(namespace, 'title')[0];
                    const programTitle = programTitleElement ? programTitleElement.textContent.trim() : "Ingen sangtittel tilgjengelig";
                    console.log('Programtittel funnet:', programTitle);

                    // Henter "Stream Title" fra annotation i track-elementet
                    const annotationContent = trackElement.getElementsByTagNameNS(namespace, 'annotation')[0]?.textContent;
                    console.log('Annotation Content mottatt:', annotationContent);

                    const streamTitle = annotationContent
                        ? annotationContent.split('\n').find(line => line.startsWith('Stream Title')).replace('Stream Title: ', '').trim()
                        : "Ingen stasjonsnavn tilgjengelig";
                    console.log('Stasjonstittel funnet:', streamTitle);

                    // Bare oppdaterer lydspilleren hvis strøm-URL har endret seg
                    const player = document.getElementById('radioPlayer');
                    if (currentStreamUrl !== streamUrl) {
                        currentStreamUrl = streamUrl; // Oppdaterer gjeldende strøm-URL
                        player.src = streamUrl;
                        player.play();
                        console.log('Spiller av ny strøm:', streamUrl);
                    }

                    // Oppdaterer sang- og stasjonsinformasjon uten å påvirke spilleren
                    const programTitleDisplay = document.getElementById('programTitle');
                    const streamTitleDisplay = document.getElementById('streamTitle');
                    programTitleDisplay.innerHTML = `<strong>Sang:</strong> ${programTitle}`;
                    streamTitleDisplay.innerHTML = `<strong>Stasjon:</strong> ${streamTitle}`;
                } else {
                    console.error('Ingen <track> element funnet.');
                }
            } catch (error) {
                console.error('Feil under innlasting av metadata:', error);
            }
        }

        // Oppdaterer metadata hvert 10. sekund uten å påvirke avspilling unødvendig
        loadRadioStream(); // Kjør umiddelbart første gang
        setInterval(loadRadioStream, 10000); // Oppdater hvert 10 sekund (10000 ms)
    </script></body>
