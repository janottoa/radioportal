const express = require('express'); const fs = require('fs'); const path = require('path'); const app = express(); const PORT = 3000; app.use(express.json()); app.use(express.static('public')); // Endepunkt for å lese stasjoner app.get('/stations', (req, res) => { const stationsPath = path.join(__dirname, 'stations.json'); fs.readFile(stationsPath, 'utf8', (err, data) => { if (err) { console.error('Feil ved lesing av stasjoner:', err); return res.status(500).send('Kunne ikke lese stasjoner'); } res.json(JSON.parse(data)); }); }); // Endepunkt for å lagre stasjoner app.post('/stations', (req, res) => { const stationsPath = path.join(__dirname, 'stations.json'); const newData = JSON.stringify(req.body, null, 2); fs.writeFile(stationsPath, newData, 'utf8', (err) => { if (err) { console.error('Feil ved lagring av stasjoner:', err); return res.status(500).send('Kunne ikke lagre stasjoner'); } res.status(200).send('Stasjoner lagret'); }); }); app.listen(PORT, () => { console.log(`Server kjører på port ${PORT}`); }); // HTML-fil for å administrere radiostasjoner const htmlContent = ` Administrer Radiostasjoner

Administrer Radiostasjoner

Stasjonsliste

`; // Serve HTML page app.get('/', (req, res) => { res.send(htmlContent); });
